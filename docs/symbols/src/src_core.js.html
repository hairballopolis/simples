 46 /**
 47  * @description used to merge objects onto the first specfied object
 48  * @param {Object} target native javascript object to be merged
 49  * @param {Object|Array} obj1, obj2.... native javascript object or array to be merged onto first
 50  **/
 51 Simples.merge = function(first /* obj1, obj2..... */ ) {
 52     // if only 1 argument is passed in assume Simples is the target
 53     var target = (arguments.length === 1 && !(this === window || this === document)) ? this: toString.call(first) === ObjectClass ? first : {};
 54     // set i to value based on whether there are more than 1 arguments
 55     var i = arguments.length > 1 ? 1: 0;
 56     // Loop over arguments
 57     for (var l = arguments.length; i < l; i++) {
 58         // if object apply directly to target with same keys
 59         var isWhat = toString.call(arguments[i]);
 60         if (isWhat === ObjectClass) {
 61             for (var key in arguments[i]) {
 62                 if (hasOwn.call(arguments[i], key)) {
 63                     target[key] = arguments[i][key];
 64                 }
 65             }
 66         } else if (isWhat === ArrayClass) {
 67             // if array apply directly to target with numerical keys
 68             push.apply(target, arguments[i]);
 69         }
 70     }
 71 
 72     return target;
 73 };

105 	/**
106 	 * @description used to add functions to browser ready queue and are triggered when DOMContentLoaded has been fired or latest window.onload
107 	 * @param {Function} callback the function to be fired when ready and or fired if already ready
108 	 */
109 	ready: function( callback ) {
110 		// Attach the listeners
111 		Simples.bindReady();
112 		
113 		// If the DOM is already ready
114 		if ( Simples.isReady ) {
115 			// Execute the function immediately
116 			callback.call( document, Simples.Event( 'ready' ) );
117 
118 		// Otherwise, remember the function for later
119 		} else if ( readyList ) {
120 			// Add the function to the wait list
121 			readyList.push( callback );
122 		}
123 	},
124 	/** @private Handle when the DOM is ready */
125 	readyHandler : function() {
126 		// Make sure that the DOM is not already loaded
127 		if ( !Simples.isReady ) {
128 			// Make sure body exists, at least, in case IE gets a little overzealous (ticket #5443).
129 			if ( !document.body ) {
130 				return setTimeout( Simples.readyHandler, 13 );
131 			}
132 
133 			// Remember that the DOM is ready
134 			Simples.isReady = true;
135 
136 			// If there are functions bound, to execute
137 			if ( readyList ) {
138 				// Execute all of them
139 				var fn, i = 0;
140 				while ( (fn = readyList[ i++ ]) ) {
141 					fn.call( document, Simples );
142 				}
143 
144 				// Reset the list of functions
145 				readyList = null;
146 			}
147 		}
148 	},
149 	/** @private To setup the event listeners for the ready event */
150 	bindReady : function(){
151 		if ( readyBound ) {
152 			return;
153 		}
154 
155 		readyBound = true;
156 
157 		// Catch cases where $(document).ready() is called after the
158 		// browser event has already occurred.
159 		if ( document.readyState === COMPLETE ) {
160 			return Simples.readyHandler();
161 		}
162 
163 		// Mozilla, Opera and webkit nightlies currently support this event
164 		if ( document.addEventListener ) {
165 			// Use the handy event callback
166 			document.addEventListener( DOMLOADED, DOMContentLoaded, false );
167 
168 			// A fallback to window.onload, that will always work
169 			window.addEventListener( "load", Simples.readyHandler, false );
170 
171 		// If IE event model is used
172 		} else if ( document.attachEvent ) {
173 			// ensure firing before onload,
174 			// maybe late but safe also for iframes
175 			document.attachEvent( READYSTATE, DOMContentLoaded);
176 
177 			// A fallback to window.onload, that will always work
178 			window.attachEvent( "onload", Simples.readyHandler );
179 
180 			// If IE and not a frame
181 			// continually check to see if the document is ready
182 			var toplevel = false;
183 
184 			try {
185 				toplevel = window.frameElement == null;
186 			} catch(e) {}
187 
188 			if ( document.documentElement.doScroll && toplevel ) {
189 				doScrollCheck();
190 			}
191 		}
192 	},
193 	/**
194 	 * @description used to set the context on a function when the returned function is executed
195 	 * @param {Object} context object to execute with
196 	 * @param {Function} func the function you want to call with the given context

237 /** @private The DOM ready check for Internet Explorer */
238 function doScrollCheck() {
239 	if ( Simples.isReady ) {
240 		return;
241 	}
242 
243 	try {
244 		// If IE is used, use the trick by Diego Perini
245 		// http://javascript.nwbox.com/IEContentLoaded/
246 		document.documentElement.doScroll(LEFT);
247 	} catch(e) {
248 		window.setTimeout( doScrollCheck, 1 );
249 		return;
250 	}
251 
252 	// and execute any waiting functions
253 	Simples.readyHandler();
254 }

256 /**
257  * @namespace Simples.fn
258  * @description the instance of a Simples object functions / instance methods
259  */
260 Simples.fn = Simples.prototype = {
261 	/**
262 	 * @constructs
263 	 * @description to initialize the Simples constructor
264 	 * @param {String|Element} selector
265 	 * @param {Object} context
266 	 * @returns {Simples} the simples onject with the results of the selector
267 	 */
268 	init : function( selector, context ){
269 
270 		// Handle $(EMPTY_STRING), $(null), or $(undefined)
271 		if ( !selector ){
272 			return this;
273 		} else if( selector.selector !== undefined ){
274 			return selector;
275 		}
276 
277 		// Handle $(DOMElement)
278 		if ( selector.nodeType || ( selector.document && selector.document.nodeType ) ) {
279 			this.context = this[0] = selector;
280 			this.length = 1;
281 			return this;
282 		}
283 		
284 		// The body element only exists once, optimize finding it
285 		if ( selector === "body" && !context ) {
286 			this.context = document;
287 			this[0] = document.body;
288 			this.selector = "body";
289 			this.length = 1;
290 			return this;
291 		}
292 	  
293 		var objClass = toString.call( selector );
294 		if( objClass === StringClass ){
295 			this.context = context;
296 			this.selector = selector;
297 		
298 			return Simples.Selector( selector, context, this );
299 
300 		} else if( objClass === HTMLCollectionClass || objClass === NodeListClass ){
301 
302 			this.push.apply( this, slice.call( selector, 0 ) );
303 	    } else if( objClass === ArrayClass ){
304 	        if( context === true ){
305 				// shortcut to use native push
306 		        this.push.apply( this, selector );
307 			} else {
308 				for(var d=0,e=selector.length;d<e;d++){
309 					if( selector[d] && ( selector[d].nodeType || selector[d].document ) ){
310 						this.push.call( this, selector[d] );
311 					}
312 				}
313 			}
314 		}
315 		return this;		
316 	},
317 	/**
318 	 * @description The count of items on the Simples object 
319 	 */
320 	length : 0,
321 	/**
322 	 * @description The selector used to create the Simples object
323 	 */
324 	selector : EMPTY_STRING,
325 	/**
326 	 * @description The version of the Simples library
327 	 */
328 	version : '@VERSION',
329 	/**
330 	 * @function each
331 	 * @description To loop over each item in the Simples object
332 	 * @param {Function} callback the function to call with each item, this is current item, arguments[ item, index, object ]
333 	 */
334 	each : [].forEach ? [].forEach : function( callback ){
335 		var i=0,l=this.length;
336 		while(i<l){
337 			callback( this[i], i, this );
338 			i++;
339 		}
340 		return this;
341 	},
342 	/**
343 	 * @description To filter the selected elements on the Simples object 
344 	 * @param {Function} callback the function to call with each item, this is current item, arguments[ item, index, object ], need to return true from callback to retain element all other return values will remove the element
345 	 */
346 	filter : function( testFn ){
347 		var i = 0,c = 0,l = this.length;
348 		while( i<l ){
349 			if( testFn( this[c], i, this ) !== true ){ 
350 				this.splice( c--, 1 );
351 			}
352 			c++; i++;
353 		}
354 		return this;
355 	},
356 	/**
357 	 * @description used to find elements off of the elements currently on the Simples object 
358 	 * @param {String} selector string to find elements
359 	 */
360 	find: function( selector ){ 
361 		var results = Simples(), i=0,l=this.length;
362 		while(i<l){
363 			Simples.Selector( selector, this[i++], results );
364 		}
365 		return results;
366 	},
367 	/**
368 	 * @description used to add more elements to the current Simples object
369 	 * @param {Elements} An array or Simples object of elements to concatenate to the current simples Object
370 	 */
371 	add : function( elems ){
372 		this.push.apply( this, slice.call( Simples( elems ), 0 ) );
373 		return this;
374 	},

376 	// Behaves like an Array's method, not like a Simples method. For hooking up to Sizzle.
377 	/** @private */
