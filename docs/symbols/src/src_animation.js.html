1 // ======= ANIMATION ========== //
  2 // Regexp used in this file
  3 var REGEX_PIXEL = /px\s?$/,
  4 	ALLOW_TYPES = /padding|margin|height|width|top|left|right|bottom|fontSize/,
  5 	TIMER_ID;
  6 /**
  7  * @namespace Simples.Animation
  8  */
  9 Simples.Animation = {
 10 	/* animations: currently active animations being run */
 11 	animations : {},
 12 	/* frameRate: global frame rate for animations */
 13 	frameRate : 24,
 14 	/* length: count of current active animations */
 15 	length : 0,
 16 	/* guid: unique identifier for animations */
 17 	guid : 1e6,
 18 	/**
 19 	 * @namespace Simples.Animation.tweens 
 20 	 * @description default tweens for animation 
 21 	 */
 22 	tweens : {
 23 		/**
 24 		 * @param frame: current frame
 25 		 * @param frameCount: total frames for animations
 26 		 * @param start: start value for tween
 27 		 * @param delta: difference to end value
 28 		*/
 29 		easing : function( frame, frameCount, start, delta ) {
 30 			return ((frame /= frameCount / 2) < 1) ? delta / 2 * frame * frame + start : -delta / 2 * ((--frame) * (frame - 2) - 1) + start;
 31 		},
 32 		/** @see Simples.Animation.tweens.easing */
 33 		linear : function( frame, frameCount, start, delta ){
 34 			return start + ( delta * ( frame/frameCount ));
 35 		},
 36 		/** @see Simples.Animation.tweens.easing */
 37 		quadratic : function( frame, frameCount, start, delta ){
 38 			return start + (((Math.cos((frame/frameCount)*Math.PI) )/2) * delta );
 39 		}
 40 	},
 41 	interval : Math.round( 1000 / this.frameRate ),
 42 	/**
 43 	 * Simples.Animation.create: used to to create an animation object which can be used by the animation queue runner
 44 	 * @param elem {Element} DOM Element to animate
 45 	 * @param setStyle {Object} CSS to use in animation, final position 
 46 	 * @param opts {Object}
 47 	 * @param opts.callback {Function} when animation complete
 48 	 * @param opts.tween {Function} tween to use when animating
 49 	 * @param opts.duration {Object} the time to elapse during animation
 50 	 */
 51 	create : function( elem, setStyle, opts ){
 52 		opts = opts || {};
 53 		if ( !( elem && elem.nodeType ) || Simples.isEmptyObject( setStyle ) ) {
 54 			if (typeof opts.callback === FUNC) {
 55 				opts.callback.call(elem);
 56 			}
 57 			return null;
 58 		}
 59 
 60 		var anim = {
 61 			0 : elem,
 62 			id : Simples.Animation.guid++,
 63 			callback : ( typeof opts.callback === FUNC ) ? opts.callback : Simples.noop,
 64 			duration : ( typeof opts.duration === NUMBER && opts.duration > -1 ) ? opts.duration : 600,
 65 			tween : ( typeof opts.tween === FUNC ) ? opts.tween : ( Simples.Animation.tweens[ opts.tween ] || Simples.Animation.tweens.easing ),
 66 			start : {},
 67 			finish : {}
 68 		};
 69 
 70 		// check for supported css animated features and prep for animation
 71 		for( var key in setStyle ){
 72 			var cKey = key.replace( RDASH_ALPHA, fcamelCase ),
 73 				opacity = ( cKey === OPACITY && setStyle[ key ] >= 0 && setStyle[ key ] <= 1 );
 74 
 75 			if( opacity || ALLOW_TYPES.test( cKey ) ){
 76 				anim.start[ cKey ] = ( Simples.getStyle( elem, cKey ) + EMPTY_STRING ).replace(REGEX_PIXEL,EMPTY_STRING) * 1;
 77 				anim.finish[ cKey ] = ( setStyle[ key ] + EMPTY_STRING ).replace(REGEX_PIXEL,EMPTY_STRING) * 1;
 78 			}                                        
 79 		}
 80 
 81 		var data = Simples.data(elem);
 82 		data.animations = data.animations || {};
 83 		data.animations[ anim.id ] = anim;
 84 
 85 		if( opts.manualStart !== true ){
 86 			Simples.Animation.start( anim );
 87 		}
 88 		return anim;
 89 	},
 90 	/**
 91 	 * Simples.Animation.start: used to add the animation to the animation runner queue
 92 	 * @param animation {Object} animation to perform action on
 93 	 */
 94 	start : function( animation ){
 95 
 96 		if( animation && animation.id ){
 97 			if( !hasOwn.call( this.animations, animation.id ) ){
 98 				this.length++;
 99 				this.animations[ animation.id ] = animation;
100 				if( animation.duration === 0 ){
101 					this.stop( animation );
102 				} else if( !animation.startTime ){
103 					animation.startTime = new Date().getTime();
104 				}
105 			}
106 			
107 			if( !TIMER_ID ){
108 				this.interval = Math.round( 1000/ this.frameRate );
109 				TIMER_ID = window.setInterval(function(){ Simples.Animation._step(); }, this.interval );
110 			}
111 		}
112 	},
113 	/**
114 	 * Simples.Animation.reverse: used to take an animation in its current position and reverse and run
115 	 * @param animation {Object} animation to perform action on
116 	 */
117 	reverse : function( animation ){
118 		var start = animation.start, finish = animation.finish;
119 
120 		animation.start = finish;
121 		animation.finish = start;
122 		
123 		if( this.animations[ animation.id ] && animation.startTime ){
124 			var now = new Date().getTime(),
125 				diff = now - animation.startTime;
126 
127 			animation.startTime = now - ( animation.duration - diff );
128 		} else {
129 			delete this.animations[ animation.id ];
130 			this.start( animation );
131 		}
132 	}, 
133 	/**
134 	 * Simples.Animation.reset: used to reset an animation to either the start or finish position
135 	 * @param animation {Object} animation to perform action on
136 	 * @param resetToEnd {Boolean} whether to reset to finish (true) or start (false||undefined) state
137 	 */
138 	reset : function( animation, resetToEnd ){
139 
140 		var cssObj = resetToEnd ? animation.finish : animation.start,
141 			elem = animation[0];
142 			
143 		for( var name in cssObj ){
144 			Simples.setStyle( elem, name, cssObj[ name ] );
145 		}
146 		
147 		if( animation.startTime ){
148 			this.stop( animation );
149 		}
150 	},
151 	/**
152 	 * @private used by the queue runner to iterate over queued animations and update each postion
153 	 */
154 	_step : function(){
155 		if( this.length ){ 
156 			var now = new Date().getTime();    
157 			for( var id in this.animations ){
158 				var anim = this.animations[ id ],
159 					diff = now - anim.startTime,
160 					elem = anim[0],
161 					duration = anim.duration;
162 					
163 				if ( diff > duration ) {
164 					this.stop( anim, true );
165 				} else {
166 					for( var name in anim.start ){
167 						var start = anim.start[ name ];
168 						Simples.setStyle( elem, name, anim.tween( diff, duration, start, anim.finish[ name ] - start ) );
169 					
170 					}
171 				}
172 			}
173 		} else if( TIMER_ID ){
174 			window.clearInterval( TIMER_ID );
175 			TIMER_ID = null;
176 		}
177 	},
178 	/**
179 	 * Simples.Animation.stop: used to stop a supplied animation and cleanup after itsef
180 	 * @param animation {Object} the animation object to use and work on.
181 	 * @param jumpToEnd {Boolean} whether to leave in current position or set css to finish position
182 	 */
183 	stop : function( animation, jumpToEnd ){
184 		if( animation && hasOwn.call( this.animations, animation.id ) ){
185 
186 			animation.startTime = null;
187 
188 			if ( jumpToEnd ){
189 				this.reset( animation, true );
190 			}
191 
192 			var data = Simples.data( animation[0] );
193 			data.animations = data.animations || {};
194 			delete data.animations[ animation.id ];
195 
196 			animation.callback.call(animation[0], animation);
197 			delete this.animations[ animation.id ];
198 			this.length--;
199 		}
200 	}
201 };

203 Simples.extend( /** @lends Simples.fn */ {
204 	/**
205 	 * @description From the instance of the Simples object used to bridge to the Simples.Animation functionality
206 	 * @param action {String} the name of the action to be performed, excluding create && _step
207 	 */
208 	animations: function(action) {
209 		if( action != ("create" || "_step") && Simples.Animation[ action ] ){
210 			var i = this.length;
211 			while (i) {
212 				var anims = Simples.data( this[--i], "animation" );
213 				Simples.Animation[ action ]( anim, arguments[2] );
214 			}
215 		}
216 		return this;
217 	},
218 	/**
219 	 * @description Used to create animations off the elements in the instance of the Simples object
220 	 * @param action {String} the name of the action to be performed, excluding create && _step
221 	 * @param css {Object} CSS to use in animation, final position 
222 	 * @param opts {Object}
223 	 * @param opts.callback {Function} when animation complete
224 	 * @param opts.tween {Function} tween to use when animating
225 	 * @param opts.duration {Object} the time to elapse during animation
226 	 */
227 	animate: function(css, opts) {
228 		var i = this.length;
229 		while (i) {
230 			Simples.animate( this[--i], css, opts );
231 		}
232 		return this;
233 	}
