1 // Constants
  2 var SINGLE_TAG = /<(\w+)\s?\/?>/,
  3 	// TAG_STRIP = /\b[\.|\#\|\[].+/g, 
  4 	// TAG_STRIP = /\b(\.|\#|\[)|(\=?<!(name))(.)*/, /(?:\w+)\b((\.|\#|\[)|(\=?>!(name)))(.)*/, /(?:\w+)\b[\.|\#|\[]{1}.*/g,
  5 	FIRST_ID = /\s#/,
  6 	// ATTR_NAME_IS = /\[name\=([^\]]+)\]/,
  7 	// Is it a simple selector
  8 	COMPLEX_TAG = /^<([a-zA-Z][a-zA-Z0-9]*)([^>]*)>(.*)<\/\1>/i,
  9 	SPACE_WITH_BOUNDARY = /\b\s+/g,
 10 	COMMA_WITH_BOUNDARY = /\s?\,\s?/g,
 11 	QUERY_SELECTOR = Simples.support.useQuerySelector,
 12 	/**
 13 	 * @private
 14 	 */
 15 	getElements = function(selector, context) {
 16 
 17 	    context = context || document;
 18 	    var tag = selector.substring(1),
 19 	    elems,
 20 	    nodes;
 21 
 22 	    if (selector.indexOf('#') === 0) {
 23 	        // Native function
 24 	        var id = (context && context.nodeType === 9 ? context: document).getElementById(tag);
 25 	        // test to make sure id is the own specified, because of name being read as id in some browsers
 26 	        return id && id.id === tag ? [id] : [];
 27 
 28 	    } else if (selector.indexOf('.') === 0) {
 29 	        if (context.getElementsByClassName) {
 30 	            // Native function
 31 	            return slice.call(context.getElementsByClassName(tag), 0);
 32 	        } else {
 33 	            // For IE which doesn't support getElementsByClassName
 34 	            elems = context.getElementsByTagName('*');
 35 	            nodes = [];
 36 	            // Loop over elements to test for correct class
 37 	            for (var i = 0, l = elems.length; i < l; i++) {
 38 	                // Detect whether this element has the class specified
 39 	                if ( Simples.className( elems[i], tag ) ) {
 40 	                    nodes.push(elems[i]);
 41 	                }
 42 	            }
 43 	            return nodes;
 44 	        }
 45 	    } else if (selector.indexOf('[name=') === 0) {
 46 	        var name = selector.substring(6).replace(/\].*/, EMPTY_STRING);
 47 	        context = context && context.nodeType === 9 ? context: document;
 48 	        if (context.getElementsByName) {
 49 	            return slice.call(context.getElementsByName(name));
 50 	        } else {
 51 	            // For IE which doesn't support getElementsByClassName
 52 	            elems = context.getElementsByName('*');
 53 	            nodes = [];
 54 	            // Loop over elements to test for correct class
 55 	            for (var m = 0, n = elems.length; m < n; m++) {
 56 	                // Detect whether this element has the class specified
 57 	                if ((" " + (elems[m].name || elems[m].getAttribute("name")) + " ").indexOf(name) > -1) {
 58 	                    nodes.push(elems[m]);
 59 	                }
 60 	            }
 61 	            return nodes;
 62 	        }
 63 	    } else {
 64 	        // assume that if not id or class must be tag
 65 	        var find = context.getElementsByTagName(selector);
 66 	        return find ? slice.call(find, 0) : [];
 67 	    }
 68 	},
 69 	/**
 70 	 * @private provide html & object to bind to
 71 	 */
 72 	createDOM = function( selector, results ){
 73 
 74 		results.context = document;
 75 
 76 		if( COMPLEX_TAG.test( selector ) ){
 77             results.selector = "<"+COMPLEX_TAG.exec( selector )[1]+">";
 78 
 79 			var div = document.createElement('div');
 80             div.innerHTML = selector;
 81             results.push.apply( results, slice.call( div.childNodes, 0 ) );
 82 
 83         } else if( SINGLE_TAG.test( selector ) ) {
 84             var tag = SINGLE_TAG.exec( selector );
 85 
 86 			results.selector = tag[0];
 87             results.push( document.createElement(tag[1]) );
 88 		}
 89 
 90 		return results;
 91 	};

 93 /**
 94  * Simples#Selector: used to create or select Elements selector based on .class #id and [name=name]
 95  * @param {String|Element} selector element is used by object and string is used to select Element(s), based on className, id and name and where the querySelector is available using querySelectorAll
 96  * @param {Element} context element used to provide context
 97  * @param {Object|Array} results optional object to return selected Elements
 98  */
 99 Simples.Selector = function(selector, context, results) {
100     results = results || [];
101 	results.selector = selector;
102 	results.context = context || document;
103 
104     if (typeof(selector) === STRING) {
105         // check selector if structured to create element
106 		if( selector.indexOf('<') > -1 && selector.indexOf('>') > 0 ){
107 			return createDOM( selector, results );
108         } else if ( QUERY_SELECTOR ) {
109             results.push.apply(results, slice.call((context || document).querySelectorAll(selector), 0));
110             return results;
111         } else {
112 	        // if it is a multi select split and short cut the process
113 	        if (COMMA_WITH_BOUNDARY.test(selector)) {
114 	            var get = selector.split(COMMA_WITH_BOUNDARY);
115 
116 	            for (var x = 0, y = get.length; x < y; x++) {
117 
118 	                results.push.apply(results, slice.call(Simples.Selector(get[x], context), 0));
119 	            }
120 	            return results;
121 	        }
122             // clean up selector
123             // selector = selector.replace(TAG_STRIP, EMPTY_STRING);
124             // get last id in selector
125             var index = selector.lastIndexOf(FIRST_ID);
126             selector = selector.substring(index > 0 ? index: 0);
127             // allow another document to be used for context where getting by id
128             results.context = context = (selector.indexOf('#') === 0 || selector.indexOf('[name=') === 0) ? (context && context.nodeType === 9 ? context: document) : (context || document);
129             var split = selector.split(SPACE_WITH_BOUNDARY);
130 
131             for (var i = 0, l = split.length; i < l; i++) {
132                 if (context.length > 0) {
133 
134                     var result = [];
135                     for (var m = 0, n = context.length; m < n; m++) {
136 
137                         result = result.concat(getElements(split[i], context[m]));
138                     }
139                     context = result;
140                 } else {
141                     context = getElements(split[i], context);
142                 }
143             }
144             results.push.apply(results, context);
145         }
146     }
147     return results;
148 };
